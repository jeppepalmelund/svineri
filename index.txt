    <label>Din årlige indkomst før skat (DKK)
      <input id="income" type="number" value="500000" min="0" step="1000">

    <button id="calc">Beregn</button>

    <div id="output" class="result" aria-live="polite" hidden></div>

    <p><small>Antagelser: Danmark har ca. 4,6 mio. skatteydere. Den samlede landbrugsstøtte på 7 mia. kr. dækker alle sektorer, hvoraf ca. 40 % tilskrives svineproduktion og foder.</small></p>

<script>
  function formatDKK(n){
    return n.toLocaleString('da-DK',{style:'currency',currency:'DKK',maximumFractionDigits:0});
  }

  document.getElementById('calc').addEventListener('click',()=>{
    const income = Number(document.getElementById('income').value)||0;
    const total_subsidy = 7000000000; // 7 mia DKK
    const pig_share = 0.4; // 40 %
    const taxpayers = 4600000; // ca. antal skatteydere

    const pig_subsidy = total_subsidy * pig_share;
    const subsidy_per_taxpayer = total_subsidy / taxpayers;
    const pig_subsidy_per_taxpayer = pig_subsidy / taxpayers;

    const pct_of_income = income>0 ? (pig_subsidy_per_taxpayer/income)*100 : 0;

    const out = document.getElementById('output');
    out.hidden = false;
    out.innerHTML = `
      <h3>Resultat</h3>
      <p>Samlet dansk landbrugsstøtte: <strong>${formatDKK(total_subsidy)}</strong></p>
      <p>Andel til svineproduktion og foder (40 %): <strong>${formatDKK(pig_subsidy)}</strong></p>
      <p>Din andel af landbrugsstøtten: <strong>${formatDKK(subsidy_per_taxpayer)}</strong></p>
      <p>Heraf går til svineproduktion og foder: <strong>${formatDKK(pig_subsidy_per_taxpayer)}</strong> (<strong>${pct_of_income.toFixed(3)} %</strong> af din indkomst)</p>
    `;
  });
</script>
